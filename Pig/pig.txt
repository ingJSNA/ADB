
GroupByOccupation = GROUP users BY occupation;

CountByOccupation = FOREACH GroupByOccupation GENERATE CONCAT((chararray)$0,CONCAT(':',(chararray)COUNT($1)));

STORE CountByOccupation INTO 'pig_output_count_occupations' USING PigStorage('\t');
